    while(t--){
        int n, c, d;
        cin>>n>>c>>d;
        unordered_map<int, int> mp;
        int val;
        int sm = INT_MAX;
        vector<vector<int>> vc(n, vector<int> (n, 0));

            mp[val]++;
            sm = min(sm, val);

        for(int i=1; i<n; i++){

                    val = vc[i-1][j] + d;
                    if(mp.find(val) == mp.end() || mp[val] == 0){
                        ans = 0;

                    }
                    mp[val] -= 1;
                    vc[i][j] = val;

            }
        }
