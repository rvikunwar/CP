int gcd(int a, int b){
    if(b == 0){
        return a;
    }
    return gcd(b, a%b);
}

int main() {

    int t;

    while(t--) {
        long long n, x, y;

        long long ans = 0;


        long long lcm = (x * y) / gcd(x, y);
        int common = n/lcm;

        long long xFactors = (n/x) - common;
        long long yFactors = (n/y) - common;

        ans = (n*(n+1)/2) - ((n-xFactors)*((n-xFactors)+1)/2);

        ans -= (yFactors * (yFactors+1)/2);
